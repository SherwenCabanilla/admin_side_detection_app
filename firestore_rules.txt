rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // More secure password reset codes
    match /password_reset_codes/{email} {
      // Allow creating new codes (for password reset)
      allow create: if true;
      
      // Allow reading/updating only if:
      // 1. User is authenticated AND
      // 2. The email matches their email OR they're verifying their own code
      allow read, update: if request.auth != null && 
        (request.auth.token.email == email || 
         resource.data.email == request.auth.token.email);
    }
    
    // User data - allow admin operations
    match /users/{userId} {
      // Allow users to access their own data
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Allow admin operations (for development - you can restrict this later)
      allow read, write: if true;
    }
    
    // Scan requests - authenticated users only
    match /scan_requests/{requestId} {
      allow read, write: if request.auth != null;
    }
    
    // Default rule - deny all
    match /{document=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
